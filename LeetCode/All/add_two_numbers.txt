# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        prev, curr = None, l1
        while curr:
            temp = curr.next
            curr.next = prev
            prev = curr
            curr = temp
        curr = prev
        s1 = ""
        while curr:
            s1 += str(curr.val)
            curr = curr.next
        n1 = int(s1)
        prev, curr = None, l2
        while curr:
            temp = curr.next
            curr.next = prev
            prev = curr
            curr = temp
        curr = prev
        s2 = ""
        while curr:
            s2 += str(curr.val)
            curr = curr.next
        n2 = int(s2)
        res = n1 + n2
        head2 = ListNode(int(str(res)[0]))
        curr = head2
        for i in str(res)[1:]:
            curr.next = ListNode(int(i))
            curr = curr.next
        prev, curr = None, head2
        while curr:
            temp = curr.next
            curr.next = prev
            prev = curr
            curr = temp
        return prev
